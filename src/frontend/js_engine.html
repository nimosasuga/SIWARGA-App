/** * ========================================== * S.I.W.A.R.G.A - DOM ENGINE * Pure DOM Manipulation (NO innerHTML) * ========================================== */ // State Global UI const UI_STATE = { isSidebarOpen: false }; /** * Merakit kerangka utama aplikasi (Sidebar, Navbar, Area Konten) */ function renderAppLayout() { const appRoot = document.getElementById('app-root'); appRoot.textContent = ''; // Bersihkan teks loading awal // 1. Buat Overlay (Background gelap saat sidebar terbuka di Mobile) const overlay = document.createElement('div'); overlay.className = 'sidebar-overlay'; overlay.id = 'sidebarOverlay'; overlay.addEventListener('click', toggleSidebar); // Tutup sidebar jika overlay diklik // 2. Buat Sidebar Container const sidebar = document.createElement('aside'); sidebar.className = 'sidebar'; sidebar.id = 'appSidebar'; // 2a. Header Sidebar (Logo/Judul) const sidebarHeader = document.createElement('div'); sidebarHeader.style.padding = '20px'; sidebarHeader.style.borderBottom = '1px solid var(--border-color)'; sidebarHeader.style.fontWeight = 'bold'; sidebarHeader.style.fontSize = '1.3rem'; sidebarHeader.style.color = 'var(--primary)'; sidebarHeader.textContent = 'S.I.W.A.R.G.A'; sidebar.appendChild(sidebarHeader); // 2b. Wadah Menu Dinamis (Kosongkan dulu, nanti diisi via RBAC) const menuContainer = document.createElement('nav'); menuContainer.id = 'menuContainer'; menuContainer.style.flex = '1'; menuContainer.style.padding = '10px'; menuContainer.style.overflowY = 'auto'; sidebar.appendChild(menuContainer); // 3. Buat Main Content Area const mainContent = document.createElement('main'); mainContent.className = 'main-content'; mainContent.id = 'mainContent'; // 3a. Navbar Atas (Header Area) const topNavbar = document.createElement('header'); topNavbar.style.height = 'var(--header-height)'; topNavbar.style.backgroundColor = '#fff'; topNavbar.style.borderBottom = '1px solid var(--border-color)'; topNavbar.style.display = 'flex'; topNavbar.style.alignItems = 'center'; topNavbar.style.padding = '0 20px'; // Tombol Hamburger (Khusus Mobile) const menuBtn = document.createElement('button'); menuBtn.innerHTML = '☰'; // Karakter Unicode Hamburger menuBtn.style.background = 'none'; menuBtn.style.border = 'none'; menuBtn.style.fontSize = '1.5rem'; menuBtn.style.cursor = 'pointer'; menuBtn.style.color = 'var(--text-main)'; // Sembunyikan tombol di layar desktop (lebar > 768px) if (window.innerWidth > 768) { menuBtn.style.display = 'none'; } menuBtn.addEventListener('click', toggleSidebar); topNavbar.appendChild(menuBtn); mainContent.appendChild(topNavbar); // 3b. Wadah Halaman Dinamis (Tempat tabel/form akan dirender) const pageContainer = document.createElement('div'); pageContainer.id = 'pageContainer'; pageContainer.style.padding = '20px'; pageContainer.textContent = 'Modul Login & Dashboard akan dimuat di sini...'; mainContent.appendChild(pageContainer); // 4. Injeksi semua ke Root (Menyusun DOM) appRoot.appendChild(overlay); appRoot.appendChild(sidebar); appRoot.appendChild(mainContent); } /** * Logika buka/tutup Sidebar (Fluid Animation) */ function toggleSidebar() { const sidebar = document.getElementById('appSidebar'); const overlay = document.getElementById('sidebarOverlay'); UI_STATE.isSidebarOpen = !UI_STATE.isSidebarOpen; if (UI_STATE.isSidebarOpen) { sidebar.classList.add('active'); overlay.classList.add('active'); } else { sidebar.classList.remove('active'); overlay.classList.remove('active'); } } // Booting DOM saat HTML selesai diload oleh Browser window.onload = () => { renderAppLayout(); };
